Assemblies

If you want to self host your web application or write a NetFluid server you need to setup the NetFluid engine.
The current assembly is automatically loaded as default web application, any other assembly loaded through the method Engine.Load is added to default web application.
To add an assembly under a specific virtual host, use the method Engine.LoadHost

Interfaces and public folders

Any NetFluid settings is loadable by app configuration using Engine.LoadAppConfiguration or programmatically specifying any settings way Engine methods 

        public static void Main()
        {
            //Add mywebapp.dll to the default web application
            Engine.Load(Assembly.LoadFile("mywebapp.dll"));

            //Load adminApp.dll into the virtual host admin.netfluid.org
            Engine.LoadHost("admin.netfluid.org", Assembly.LoadFile("adminApp.dll"));

            //Virtual host www.supb.eu is FastCGI fowarded to apache.netfluid.org on port 5000
            Engine.AddFowarding("www.supb.eu", "apache.netfluid.org:5000");

            //Load netfluid setting from App.config file, return false if missing or invalid
            if (!Engine.LoadAppConfiguration())
            {
                //Add an http interface on any available public ip on specified port
                Engine.Interfaces.AddAllAddresses(80);

                //Add an http interface on ip 127.0.0.1
                Engine.Interfaces.AddLoopBack(8080);

                //Any files inside "./myfiles/public" folders appears under the Uri path /public
                Engine.AddPublicFolder("/public", "./myfiles/public");
            }

            //Engine start to work
            Engine.Start();

            Console.ReadLine();
        }
